# ROOT CMAKELISTS.TXT
# INIT
cmake_minimum_required(VERSION 3.13)

set(INPUT_NAME "prog" CACHE STRING '')

project(
    ${INPUT_NAME}
)

#include dir
set(
    INCLUDE_DIRS
    "src"
)


# Ctest
include(CTest)
enable_testing()

# sources cache
set(${PROJECT_NAME}_SOURCES "" CACHE INTERNAL "")

list(APPEND CMAKE_MODULE_PATH "${CMAKE_CURRENT_LIST_DIR}/cmake/modules")

include(CMAKE_PROJECT_GET_SOURCES_functions)

#   Find threads num
include(ProcessorCount)
ProcessorCount(core_num)
message("<Working on ${core_num} threads processor>/n")

# CODE 
say_path_to_root_cmake()

#   Create the main exe
message("")


add_executable(
    ${PROJECT_NAME}
)

foreach(dir ${INCLUDE_DIRS})
    add_subdirectory(${dir})
endforeach()
